```javascript
// 现代化状态管理
import { defineStore } from 'pinia'

export const useProjectStore = defineStore('project', {
  state: () => ({
    projects: [],
    currentProject: null,
    loading: false,
    pagination: {
      page: 1,
      size: 10,
      total: 0
    }
  }),
  
  getters: {
    // 计算属性
    completedProjects: (state) => 
      state.projects.filter(p => p.status === 'completed'),
    
    projectStats: (state) => ({
      total: state.projects.length,
      completed: state.projects.filter(p => p.status === 'completed').length,
      inProgress: state.projects.filter(p => p.status === 'in_progress').length
    })
  },
  
  actions: {
    // 异步操作
    async fetchProjects(params = {}) {
      this.loading = true
      try {
        const { data } = await api.getProjects({
          ...this.pagination,
          ...params
        })
        this.projects = data.list
        this.pagination.total = data.total
      } finally {
        this.loading = false
      }
    },
    
    // 本地操作
    addProject(project) {
      this.projects.unshift(project)
    },
    
    updateProject(updatedProject) {
      const index = this.projects.findIndex(p => p.id === updatedProject.id)
      if (index !== -1) {
        this.projects.splice(index, 1, updatedProject)
      }
    }
  }
})
```

```javascript
// 性能优化工具类
class PerformanceOptimizer {
  // 图片懒加载
  static lazyLoadImages() {
    const images = document.querySelectorAll('img[data-src]')
    const observer = new IntersectionObserver((entries) => {
      entries.forEach(entry => {
        if (entry.isIntersecting) {
          const img = entry.target
          img.src = img.dataset.src
          img.removeAttribute('data-src')
          observer.unobserve(img)
        }
      })
    })
    
    images.forEach(img => observer.observe(img))
  }
  
  // 请求防抖
  static debounce(fn, delay) {
    let timer = null
    return function(...args) {
      clearTimeout(timer)
      timer = setTimeout(() => {
        fn.apply(this, args)
      }, delay)
    }
  }
  
  // 请求节流
  static throttle(fn, interval) {
    let lastTime = 0
    return function(...args) {
      const now = Date.now()
      if (now - lastTime >= interval) {
        fn.apply(this, args)
        lastTime = now
      }
    }
  }
  
  // 缓存管理
  static createCache(maxSize = 100) {
    const cache = new Map()
    
    return {
      get(key) {
        if (cache.has(key)) {
          const value = cache.get(key)
          // 移动到最近使用
          cache.delete(key)
          cache.set(key, value)
          return value
        }
        return null
      },
      
      set(key, value) {
        if (cache.has(key)) {
          cache.delete(key)
        } else if (cache.size >= maxSize) {
          // 删除最久未使用
          const firstKey = cache.keys().next().value
          cache.delete(firstKey)
        }
        cache.set(key, value)
      }
    }
  }
}
```

#### 7. 前端监控体系
```javascript
// 前端监控SDK
class FrontendMonitor {
  constructor(options = {}) {
    this.apiUrl = options.apiUrl || '/api/monitor'
    this.sampleRate = options.sampleRate || 0.1
    this.queue = []
    this.isSending = false
  }
  
  // 错误监控
  initErrorMonitor() {
    window.addEventListener('error', (event) => {
      this.trackError({
        type: 'js_error',
        message: event.message,
        filename: event.filename,
        lineno: event.lineno,
        colno: event.colno,
        stack: event.error?.stack
      })
    })
    
    window.addEventListener('unhandledrejection', (event) => {
      this.trackError({
        type: 'promise_error',
        reason: event.reason?.toString()
      })
    })
  }
  
  // 性能监控
  initPerformanceMonitor() {
    if (window.performance) {
      const timing = performance.timing
      const metrics = {
        dns: timing.domainLookupEnd - timing.domainLookupStart,
        tcp: timing.connectEnd - timing.connectStart,
        ttfb: timing.responseStart - timing.requestStart,
        dom: timing.domContentLoadedEventEnd - timing.domLoading,
        load: timing.loadEventEnd - timing.navigationStart
      }
      
      this.trackPerformance(metrics)
    }
  }
  
  // 用户行为监控
  initUserBehaviorMonitor() {
    // PV/UV统计
    this.trackPageView()
    
    // 点击热力图
    document.addEventListener('click', (event) => {
      const element = event.target
      const path = this.getElementPath(element)
      const position = { x: event.clientX, y: event.clientY }
      
      this.trackClick({ path, position })
    }, true)
  }
  
  // 发送数据
  send(data) {
    if (Math.random() > this.sampleRate) return
    
    this.queue.push({
      ...data,
      timestamp: Date.now(),
      url: window.location.href,
      userAgent: navigator.userAgent
    })
    
    this.flush()
  }
  
  // 批量发送
  flush() {
    if (this.isSending || this.queue.length === 0) return
    
    this.isSending = true
    const batch = this.queue.splice(0, 20)
    
    fetch(this.apiUrl, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(batch)
    }).finally(() => {
      this.isSending = false
      if (this.queue.length > 0) {
        setTimeout(() => this.flush(), 1000)
      }
    })
  }
}

// 使用监控
const monitor = new FrontendMonitor()
monitor.initErrorMonitor()
monitor.initPerformanceMonitor()
monitor.initUserBehaviorMonitor()
```

```javascript
// XSS防护
class XSSProtection {
  static sanitize(input) {
    if (typeof input !== 'string') return input
    
    const element = document.createElement('div')
    element.textContent = input
    return element.innerHTML
  }
  
  static validateUrl(url) {
    try {
      const parsed = new URL(url)
      const allowedProtocols = ['http:', 'https:', 'mailto:', 'tel:']
      
      if (!allowedProtocols.includes(parsed.protocol)) {
        throw new Error('不支持的协议')
      }
      
      return parsed.toString()
    } catch {
      return null
    }
  }
}

// CSRF防护
class CSRFProtection {
  static getToken() {
    let token = localStorage.getItem('csrf_token')
    
    if (!token) {
      token = this.generateToken()
      localStorage.setItem('csrf_token', token)
    }
    
    return token
  }
  
  static generateToken() {
    const array = new Uint8Array(32)
    window.crypto.getRandomValues(array)
    return Array.from(array, byte => byte.toString(16).padStart(2, '0')).join('')
  }
  
  static validateToken(token) {
    const storedToken = localStorage.getItem('csrf_token')
    return token && storedToken && token === storedToken
  }
}

// 使用安全防护
const safeInput = XSSProtection.sanitize(userInput)
const safeUrl = XSSProtection.validateUrl(userUrl)

// 请求时添加CSRF Token
axios.interceptors.request.use(config => {
  config.headers['X-CSRF-Token'] = CSRFProtection.getToken()
  return config
})
```
