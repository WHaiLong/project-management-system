```bash
GITLAB_HASH=$(git ls-remote gitlab main 2>/dev/null | awk '{print $1}')

echo "GitHub: $GITHUB_HASH"
echo "GitLab: $GITLAB_HASH"

if [ "$GITHUB_HASH" = "$GITLAB_HASH" ]; then
    echo "✅ 代码同步成功"
else

fi
```
