   export const useAdvancedStore = defineStore('advanced', () => {
     // 响应式状态
     const state = reactive({
       data: [] as DataItem[],
       loading: false,
       error: null as string | null
     })
     
     // 计算属性
     const stats = computed(() => ({
       total: state.data.length,
       active: state.data.filter(item => item.status === 'active').length,
       completed: state.data.filter(item => item.status === 'completed').length
     }))
     
     // 异步操作
     async function fetchData(params: FetchParams) {
       state.loading = true
       state.error = null
       
       try {
         const response = await api.fetchData(params)
         state.data = response.data
         
         // 缓存到本地存储
         localStorage.setItem('cached_data', JSON.stringify(state.data))
       } catch (error) {
         state.error = error.message
         
         // 降级策略：使用缓存数据
         const cached = localStorage.getItem('cached_data')
         if (cached) {
           state.data = JSON.parse(cached)
         }
       } finally {
         state.loading = false
       }
     }
     
     // 批量操作
     function batchUpdate(updates: BatchUpdate[]) {
       updates.forEach(update => {
         const index = state.data.findIndex(item => item.id === update.id)
         if (index !== -1) {
           Object.assign(state.data[index], update.changes)
         }
       })
       
       // 防抖保存
       debouncedSave()
     }
     
     // 返回所有内容
     return { ...toRefs(state), stats, fetchData, batchUpdate }
   })
